面向对象

Go语言不是一门面向对象编程语言，使用组合的方式，通过接口和结构体模拟面向对象。

接口

type Animal interface {
	Sleep()
}

type Mouse struct {
	Name string
}

func NewMouse(name string) Animal {
	return &Mouse{
		Name: name,
	}
}

func (m *Mouse) Sleep() {
	fmt.Println(m.Name + "正在睡")
}

用组合代替继承

type Animal struct {
	Name string
}

func (a *Animal) Sleep() {
	fmt.Println(a.Name + "正在睡")
}

type Mouse struct {
	Animal
}

func NewMouse(name string) *Mouse {
	return &Mouse{
		Animal: Animal{
			Name: name,
		},
	}
}

反射

反射就是动态的获取变量类型信息和值信息的机制。应用场景如下：

  • 序列化，例如：google.golang.org/grpc
  • 数据库ORM，例如：github.com/jinzhu/gorm
  • 配置文件解析，例如：github.com/BurntSushi/toml

一、对变量进行处理

获取变量的类型信息

var x = "hello world"

t := reflect.TypeOf(x)
fmt.Printf("type of a is:%v\n", t)

k := t.Kind()
switch k {
case reflect.Int:
    fmt.Printf("a is int\n")
case reflect.String:
    fmt.Printf("a is string\n")
}

获取变量的值信息

var x = "hello world"

v := reflect.ValueOf(x)
fmt.Printf("type of a is:%v\n", v.Type())

k := v.Kind()
switch k {
case reflect.Int:
    fmt.Printf("a is int, store value is:%d\n", v.Int())
case reflect.String:
    fmt.Printf("a is string, store value is:%s\n", v.String())
}

对值进行赋值操作

var x = 3.14

v := reflect.ValueOf(&x) // 这里传地址
v.Elem().SetFloat(6.28)

二、对结构体进行处理

获取结构体的字段

可以通过 NumField() 获取所有结构体字段的数目、进而遍历，通过Field()方法获取字段的信息。

x := Student{
    Name:  "BigOrange",
    Sex:   1,
    Age:   10,
    Score: 80.1,
}

v := reflect.ValueOf(x)
t := v.Type()

// NumFiled()获取字段数，v.Field(i)可以取得下标位置的字段信息，返回的是一个Value类型的值
for i := 0; i < v.NumField(); i++ {
    field := v.Field(i)
    // 打印字段的名称、类型以及值
    fmt.Println(t.Field(i).Name, field.Type().Kind(), field.Interface())
}

对结构体内的字段进行赋值操作

x := Student{
    Name:  "BigOrange",
    Sex:   1,
    Age:   10,
    Score: 80.1,
}

v := reflect.ValueOf(&x) // 这里传地址

v.Elem().Field(0).SetString("ChangeName") // 通过索引赋值
v.Elem().FieldByName("Score").SetFloat(99.9) // 通过字段名称赋值

获取结构体里面的方法

x := Student{
    Name:  "BigOrange",
    Sex:   1,
    Age:   10,
    Score: 80.1,
}

v := reflect.ValueOf(&x)
t := v.Type()

for i := 0; i < t.NumMethod(); i++ {
    method := t.Method(i)
    fmt.Println(i, method.Name, method.Type)
}

获取结构体里tag

x := Student{
    Name: "BigOrange",
}

v := reflect.ValueOf(&x)
t := v.Type()

field0 := t.Elem().Field(0) // 通过索引获取字段
// field0, _ := t.Elem().FieldByName("Name") // 通过字段名称获取字段

fmt.Println(field0.Tag.Get("json"))

错误处理机制

golang中一个协程发生panic而没有被捕获，会导致整个程序终止。java中一个线程发生异常，只要其主线程不终止，那么整个程序还是运行的。

通过defer和recover捕获异常：

defer func () {
    if err := recover(); err != nil {
        fmt.Println(err)
    }
}()

捕获panic异常

func main() {
	router := gin.New()
	router.Use(gin.LoggerWithFormatter(func(param gin.LogFormatterParams) string {
		return fmt.Sprintf("%s - [%s] \"%s %s %s %d %s \"%s\" %s\"\n",
			param.ClientIP,
			param.TimeStamp.Format(time.RFC1123),
			param.Method,
			param.Path,
			param.Request.Proto,
			param.StatusCode,
			param.Latency,
			param.Request.UserAgent(),
			param.ErrorMessage,
		)
	}))
	router.Use(gin.Recovery()) // 捕获和处理异常

	router.GET("/ping", func(c *gin.Context) {
		c.String(200, "pong")
	})

	router.Run(":5000")
}

泛型

最早在Go 1.17版本支持，目前可通过如下满足一部分场景。

type Sortable interface {
	Len() int
	Less(int, int) bool
	Swap(int, int)
}

func bubbleSort(array Sortable) {
	for i := 0; i < array.Len(); i++ {
		for j := 0; j < array.Len()-i-1; j++ {
			if array.Less(j+1, j) {
				array.Swap(j, j+1)
			}
		}
	}
}

// 实现接口的整型切片
type IntArr []int

func (array IntArr) Len() int {
	return len(array)
}

func (array IntArr) Less(i int, j int) bool {
	return array[i] < array[j]
}

func (array IntArr) Swap(i int, j int) {
	array[i], array[j] = array[j], array[i]
}

// 实现接口的字符串，按照长度排序
type StringArr []string

func (array StringArr) Len() int {
	return len(array)
}

func (array StringArr) Less(i int, j int) bool {
	return len(array[i]) < len(array[j])
}

func (array StringArr) Swap(i int, j int) {
	array[i], array[j] = array[j], array[i]
}

// 测试
func main() {
	intArray1 := IntArr{3, 4, 2, 6, 10, 1}
	bubbleSort(intArray1)
	fmt.Println(intArray1)

	stringArray1 := StringArr{"hello", "i", "am", "go", "lang"}
	bubbleSort(stringArray1)
	fmt.Println(stringArray1)
}

协程

协程是一种用户态的轻量级线程。不要以共享内存的方式来通信，相反，要通过通信来共享内存。

信道通信

func main() {
	ch := make(chan int)
	done := make(chan bool)

	// 写
	go func() {
		for {
			// select语句，每个case必须是一个通信操作，要么是发送要么是接收，随机执行一个可运行的case。
			select {
			case ch <- rand.Int(): // 发送消息
			}
		}
	}()

	// 读
	go func() {
		for i := 0; i < 5; i++ {
			fmt.Println("rand number = ", <-ch)
		}
		done <- true // 发送终止信号并返回
		return
	}()

	<-done // 接收到终止信号时退出主程序
}

消息队列

func main() {
	c := make(chan int, 2)

	// 消费者
	go func() {
		for m := range c {
			fmt.Println(m)
		}
	}()

	// 消费者
	go func() {
		for m := range c {
			fmt.Println(m)
		}
	}()

	// 生产者
	go func() {
		for i := 0; i < 10; i++ {
			c <- i
		}
	}()

	time.Sleep(time.Second)
}

超时处理

func main() {
	ch := make(chan bool, 1)
	go func() {
		time.Sleep(time.Second * 2)
		ch <- true
	}()

	select {
	case res := <-ch:
		fmt.Println(res)
	case <-time.After(time.Second * 3):
		fmt.Println("timeout")
	}
}

func main() {
	c := make(chan int)

	var wg sync.WaitGroup
	for i := 0; i < 10; i++ {
		wg.Add(1)
		go func() {
			c <- 1
			wg.Done()
		}()
	}

	// 等待关闭channel
	go func() {
		wg.Wait()
		close(c)
	}()

}

sync

sync.Atomic

sync/atomic 可以在并发场景下对变量进行非侵入式的操作。可以保证并发安全，虽然使用sync.Mutex可以实现，但是使用sync/atomic不仅是轻量级的，而且代码也更加简洁。

func main() {
    count := int64(0)

	var wg sync.WaitGroup
	for i := 0; i < 10; i++ {
		wg.Add(1)
		go func(i int) {
			atomic.AddInt64(&count, 1) // 原子操作
			wg.Done()
		}(i)
	}
	wg.Wait()
}

sync.Mutex

互斥锁和读写互斥锁。

解决map并发不安全

func main() {
	var lock sync.Mutex
	m := make(map[int]int)

	go func() {
		for i := 0; i < 10; i++ {
			lock.Lock()
			m[i] = i
			lock.Unlock()
		}
	}()

	go func() {
		for i := 0; i < 10; i++ {
			lock.Lock()
			fmt.Println(m[i])
			lock.Unlock()
		}
	}()
}

解决slice并发不安全

var s []int
var lock sync.Mutex

func appendValue(i int) {
	lock.Lock()
	s = append(s, i)
	lock.Unlock()
}

func main() {
	for i := 0; i < 10; i++ {
		go appendValue(i)
	}
}

在读多写少的环境中，可以优先使用读写互斥锁（sync.RWMutex），它比互斥锁更加高效。

sync.Once

sync.Once是Golang package中使方法只执行一次的对象实现。

func main() {
	var once sync.Once
	flag := make(chan bool)

	for i := 0; i < 10; i++ {
		go func() {
			once.Do(func() {
				fmt.Println("only once")
			})
			flag <- true
		}()
	}

	for i := 0; i < 10; i++ {
		<-flag
	}
}

sync.Pool

sync.Pool是一个临时的对象池，适用于储存一些会在goroutine间共享的临时对象，减少内存的分配。

var bufPool = sync.Pool {
	New: func() interface{} {
		return new(bytes.Buffer)
	},
}

func main()  {
	buf := bufPool.Get().(*bytes.Buffer)
	buf.Reset()
	buf.WriteString("hello")
	bufPool.Put(buf)
}

sync.Pool缓存的期限只是两次gc之间这段时间

sync.WaitGroup

阻塞等待所有任务完成之后再继续执行。

func main() {
	var wg sync.WaitGroup
	for i := 0; i < 10; i++ {
		wg.Add(1)
		go func(i int) {
			defer wg.Done()
			fmt.Println(i)
		}(i)
	}
	wg.Wait()
}
